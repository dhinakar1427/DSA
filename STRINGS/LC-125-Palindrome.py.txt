class Solution:
    def isPalindrome(self, s: str) -> bool:
        left , right = 0 , len(s) -1
        while left < right :
        # move on when no alphanumeric in left side
            while left < right and not s[left].isalnum():
                left += 1
        # move on when no alphanumeric in right side        
            while left < right and not s[right].isalnum():
                right -= 1
        # compare 
            if s[left].lower() != s[right].lower():
                return False
            left += 1            
            right -= 1
        return True
